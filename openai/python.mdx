```python
from openai import OpenAI

client = OpenAI(
	base_url='https://api.mlrails.com',
	api_key='YOUR API KEY',
)

chat_completion = client.chat.completions.create(
	messages=[
		{
			"role": "user",
			"content": "Say this is a test",
		}
	],
	model="gpt-3.5-turbo",
)


print(chat_completion)

```
